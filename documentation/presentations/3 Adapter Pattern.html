Hvordan ser det ud UDEN adapter?

// PROBLEM: Logger kender BEGGE konkrete output-typer direkte
public class Logger {
    private FileLogOutputter fileLog;
    private ConsoleLogOutput consoleLog;
    private boolean useFile; // ← flag i stedet for abstraktion

    public void log(LoggerLevel level, String message) {
        if (useFile) {
            if (level == LoggerLevel.INFO)    fileLog.logInfo(message);
            else if (level == LoggerLevel.WARNING) fileLog.logWarning(message);
            else if (level == LoggerLevel.ERROR)   fileLog.logError(message);
        } else {
            consoleLog.log(level, message); // anden signatur!
        }
    }
    // Liskov: FileLogOutputter og ConsoleLogOutput er IKKE substituerbare
    // Logger MÅ kende begge typer — de kan ikke byttes ud bag et fælles interface
}
