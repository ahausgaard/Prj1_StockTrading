public class BuySharesService {
    // TÆT KOBLING: Service skal kende alle gebyrtyper
    public void buyShares(BuySharesRequest request) {
        // Hvis gebyrtype er FLAT...
        if (feeType == FeeType.FLAT) {
            BigDecimal fee = BigDecimal.valueOf(15);
        }
        // Hvis gebyrtype er PERCENTAGE...
        else if (feeType == FeeType.PERCENTAGE) {
            BigDecimal fee = totalAmount.multiply(new BigDecimal("0.01"));
        }
        // Hvis gebyrtype er PERCENTAGE_MINIMUM...
        else if (feeType == FeeType.PERCENTAGE_MINIMUM) {
            // Kompleks logik...
        }
    }
}

Testbarhed MED Adapter

Se: test/unit/business/services/trading/ (FeeStrategy tests)

// Adapter Pattern gør det muligt at teste hver gebyrtype isoleret
@Test void flatFeeStrategy_calculatesFlatFee() {
    FeeStrategy strategy = new FlatFeeStrategy(BigDecimal.valueOf(15));
    BigDecimal fee = strategy.calculateFee(new BigDecimal("100"));

    assertEquals(new BigDecimal("15"), fee);
}

@Test void percentageFeeStrategy_calculatesPercentage() {
    FeeStrategy strategy = new PercentageFeeStrategy(0.01);
    BigDecimal fee = strategy.calculateFee(new BigDecimal("1000"));

    assertEquals(new BigDecimal("10"), fee);
}

// Services kan testes med MockStrategy uden afhængigheder
@Test void buySharesService_usesInjectedFeeStrategy() {

    BuySharesService service = new BuySharesService(..., mockFee);
    // Test uden at kende konkrete gebyrtyper!
}
