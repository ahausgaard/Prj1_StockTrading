Tilstandsmaskiner uden if/else-spaghetti

Hvilket problem løser State Pattern?

  • Massive if/switch statements — Hver tilstand har egen logik
  • Uoverskueligt flow — Vanskelig at følge tilstands-overgange
  • Ufleksibelt — Tilføj ny tilstand = redigér eksisterende kode

// Massive switch uden State Pattern - uoverskueligt!
public class LiveStockBad {
    public void updatePrice() {
        switch (currentState) {
            case LIVE:
                price = price.multiply(new BigDecimal("1.03")); // +3%
                break;
            case GROWING:
                price = price.multiply(new BigDecimal("1.05")); // +5%
                if (volatility > 0.3) currentState = StockState.DECLINING;
                if (volatility < 0.05) currentState = StockState.STEADY;
                break;
            case DECLINING:
                price = price.multiply(new BigDecimal("0.95")); // -5%
                if (volatility < 0.15) currentState = StockState.STEADY;
                if (price.compareTo(BigDecimal.valueOf(0.5)) < 0)
                    currentState = StockState.BANKRUPT;
                break;
            case STEADY:
                price = price.multiply(new BigDecimal("1.01")); // +1%
                if (volatility > 0.2) currentState = StockState.GROWING;
                if (volatility < -0.2) currentState = StockState.DECLINING;
                break;
            case BANKRUPT:
                price = BigDecimal.ZERO; // Kan aldrig vende tilbage
                break;
            default:
                throw new IllegalStateException("Unknown state");
        }
    }
}

Implementering MED State

Se: src/business/stockmarket/simulation/GrowingState.java

// State Pattern: Ren logik per tilstand
public class GrowingState extends LiveStockState {
    @Override
    public void updatePrice(LiveStock context) {
        // Kun logik for GROWING tilstand
        BigDecimal newPrice = context.getCurrentPrice()
            .multiply(new BigDecimal("1.04")); // +4%
        context.setCurrentPrice(newPrice);

        // Tilstanden bestemmer selv om den skal skifte
        if (shouldTransitionToDeclining()) {
            context.setCurrentState(new DecliningState());
        }
    }

    private boolean shouldTransitionToDeclining() {
        // Simpel logik - ingen Store if/switch statements her
        return Math.random() < 0.1; // 10% chance
    }
}

Context delegerer til State

Se: src/business/stockmarket/simulation/LiveStock.java

// Context: Delegerer helt til sin state
public class LiveStock {
    private StockState currentState;
    private BigDecimal currentPrice;

    public void updatePrice() {
        // Konteksten ved ikke HVORDAN
        // Den spørger bare sin state "gør dit job"
        currentState.updatePrice(this);

        // State'n har selv opdateret prisen og eventuel skiftet til ny state
    }

    // Helpers som states kan bruge
    public void setCurrentState(StockState newState) {
        currentState = newState;
    }

    public void setCurrentPrice(BigDecimal price) {
        currentPrice = price;
    }
}

Testbarhed MED State

@Test void growingState_increasesPrice() {
    LiveStock stock = new LiveStock();
    stock.setCurrentPrice(new BigDecimal("100"));
    stock.setCurrentState(new GrowingState());

    stock.updatePrice();

    BigDecimal newPrice = stock.getCurrentPrice();
    assertTrue(newPrice.compareTo(new BigDecimal("100")) > 0);
}

@Test void decliningState_decreasesPrice() {
    LiveStock stock = new LiveStock();
    stock.setCurrentPrice(new BigDecimal("100"));
    stock.setCurrentState(new DecliningState());

    stock.updatePrice();

    BigDecimal newPrice = stock.getCurrentPrice();
    assertTrue(newPrice.compareTo(new BigDecimal("100")) < 0);
}
// Hver tilstand testes isoleret - ingen massive if/else!
